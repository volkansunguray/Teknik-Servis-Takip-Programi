CREATE TABLE dbo.tbl_personel (
  id int IDENTITY(1, 1) NOT NULL,
  personel_adi nvarchar(50) COLLATE Turkish_CI_AS NOT NULL,
  personel_eposta nvarchar(50) COLLATE Turkish_CI_AS NOT NULL,
  personel_sifre nvarchar(30) COLLATE Turkish_CI_AS NOT NULL,
  personel_telefon nvarchar(15) COLLATE Turkish_CI_AS NOT NULL,
  CONSTRAINT PK_tbl_personel PRIMARY KEY CLUSTERED (id)
    WITH (
      PAD_INDEX = OFF, IGNORE_DUP_KEY = OFF, STATISTICS_NORECOMPUTE = OFF,
      ALLOW_ROW_LOCKS = ON, ALLOW_PAGE_LOCKS = ON)
)
ON [PRIMARY]
GO

--
-- Definition for table tbl_teknikservis : 
--

CREATE TABLE dbo.tbl_teknikservis (
  id int IDENTITY(1, 1) NOT NULL,
  servis_tanim nvarchar(50) COLLATE Turkish_CI_AS NOT NULL,
  servis_aciklama nvarchar(max) COLLATE Turkish_CI_AS NOT NULL,
  baslangic_tarihi datetime CONSTRAINT DF_tbl_teknikservis_baslangic_tarihi DEFAULT getdate() NULL,
  teslim_tarihi datetime NULL,
  personel_id int NOT NULL,
  ariza_durum bit CONSTRAINT DF_tbl_teknikservis_ariza_durum DEFAULT 0 NOT NULL,
  CONSTRAINT PK_tbl_teknikservis PRIMARY KEY CLUSTERED (id)
    WITH (
      PAD_INDEX = OFF, IGNORE_DUP_KEY = OFF, STATISTICS_NORECOMPUTE = OFF,
      ALLOW_ROW_LOCKS = ON, ALLOW_PAGE_LOCKS = ON)
)
ON [PRIMARY]
GO
